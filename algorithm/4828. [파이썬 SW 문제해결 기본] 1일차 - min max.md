```python
T = int(input())

def my_max(x):
    max_output =  x[0]
    for i in range(len(x)-1):
        if max_output < x[i+1]:
            max_output = x[i+1]
    return max_output


def my_min(y):
    min_output = y[0]
    for i in range(len(y)-1):
        if min_output > y[i+1]:
            min_output = y[i+1]
    return min_output


for t in range(1, T+1):
    num_nums = int(input())
    nums = list(map(int, input().split()))
    max_sub = my_max(nums) - my_min(nums)

    print('#{} {}'.format(t, max_sub))
```

# 4831. [파이썬 S/W 문제해결 기본] 1일차 - 전기버스 

```python

T = int(input())

for t in range(1, T+1):
    inputs = list(map(int, input().split()))

    K, N, M = inputs[0], inputs[1], inputs[2]
    bus_stops = list(map(int, input().split()))

    # charge stop =1, none = 0
    charge_stops = [0] * N
    for charges in bus_stops:
        charge_stops[charges] = 1

    stop = K
    cnt = 0
    while stop < N:

        while charge_stops[stop] == 0:  # 충전소가 없다면 뒤로 가기
            stop -= 1

        cnt += 1
        stop += K

        if cnt >= N:
            break

    if cnt >= N:
        print('#{} {}'.format(t, '0'))

    else:
        print('#{} {}'.format(t, cnt))


```

# 4834. [파이썬 S/W 문제해결 기본] 1일차 - 숫자 카드

```python

T = int(input())

for t in range(1, T+1):
    num_cards = int(input())
    given_cards = list(map(int, input()))
    
    cards = [0] * 10

    for card in given_cards:
        cards[card] += 1

    largest = 0
    for idx in range(10):
        if cards[largest] <= cards[idx]:
            largest = idx


    print('#{} {} {}'.format(t, largest, cards[largest]))
```

# 4835. [파이썬 S/W 문제해결 기본] 1일차 - 구간합 

```python

T = int(input())

for t in range(1, T+1):
    inputs = list(map(int, input().split()))

    N, M = inputs[0], inputs[1]

    N_nums = list(map(int, input().split()))

    sums = []

    for i in range(N - M + 1):
        lil_sum = 0
        for j in range(i, i + M):
            lil_sum += N_nums[j]
        sums.append(lil_sum)

    print('#{} {}'.format(t,max(sums) - min(sums)))

```
